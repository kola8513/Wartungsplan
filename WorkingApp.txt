create_initial_table <- function() {
  headers <- c("TÃĊglich", "WÃ¶chentlich", "14-tÃĊgig", "Monatlich", "Bei Bedarf")
  row_counts <- c(3, 13, 3, 5, 4)
  rows <- NULL
  for (i in seq_along(headers)) {
    header <- headers[i]
    count <- row_counts[i]
    tmp <- data.frame(Header = c(header, rep("", count)), matrix("", nrow = count + 1, ncol = 31))
    colnames(tmp) <- c("Header", as.character(1:31))
    rows <- rbind(rows, tmp)
  }
  return(rows)
}

server <- function(input, output, session) {
  rv <- reactiveValues(data = create_initial_table())
  
  # Add row under section
  observeEvent(input$add_row, {
    section <- input$section
    header_idx <- which(rv$data$Header == section)
    next_header_idx <- min(which((1:nrow(rv$data)) > max(header_idx) & rv$data$Header != ""))
    if(is.infinite(next_header_idx)) next_header_idx <- nrow(rv$data) + 1
    insert_idx <- next_header_idx - 1
    new_row <- data.frame(Header = "", matrix("", nrow = 1, ncol = 31))
    colnames(new_row) <- colnames(rv$data)
    rv$data <- rbind(
      rv$data[1:insert_idx, ],
      new_row,
      rv$data[(insert_idx+1):nrow(rv$data), ]
    )
  })
  
  # Handsontable editor
  output$tableRH <- renderRHandsontable({
    rhandsontable(rv$data)
  })
  observe({
    if(!is.null(input$tableRH)) {
      rv$data <- hot_to_r(input$tableRH)
    }
  })
  
  # Save/load draft handlers
  observeEvent(input$save_draft, {
    saveRDS(rv$data, file = "draft_table.rds")
    showNotification("Draft saved.", type = "message")
  })
  observeEvent(input$load_draft, {
    if (file.exists("draft_table.rds")) {
      rv$data <- readRDS("draft_table.rds")
      showNotification("Draft loaded.", type = "message")
    } else {
      showNotification("No draft file found.", type = "error")
    }
  })
  
  # ---- CSV download (fix wrong argument) ----
  output$download_table_csv <- downloadHandler(
    filename = function() paste0("Wartungsplan_", Sys.Date(), ".csv"),
    content = function(file) {
      write.csv(rv$data, file, row.names = FALSE, fileEncoding = "UTF-8")
    },
    contentType = "text/csv"
  )
  
  # ---- PDF download (render then copy) ----
  output$download_table_pdf <- downloadHandler(
    filename = function() paste0("Wartungsplan_", Sys.Date(), ".pdf"),
    content = function(file) {
      owd <- setwd(tempdir()); on.exit(setwd(owd), add = TRUE)
      
      rmd_content <- "

```{r, echo=FALSE, results='asis'}
